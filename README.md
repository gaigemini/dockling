    mkdir gai
2. go to gai dir
    cd gai
3. start visual studio code
    code .
4. create a virtual environment on current folder
    python3 -m venv .
5. active the virtual environment
    source bin/activate
6. install depencency
    pip install -r requirements.txt
7. upgrade python library
    pip install --upgrade -r requirements.txt
